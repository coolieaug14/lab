android:layout_alignParentTop="true" android:layout_centerHorizontal="true"/>

package com.example.tenth;



import android.app.Activity;
import android.os.Bundle;
import android.os.CountDownTimer;
import android.widget.*; import android.view.View; import java.util.Random;

public class MainActivity extends Activity {


int score=0; TextView info; Button tap;
RelativeLayout root; Random r=new Random();

@Override
protected void onCreate(Bundle b){
super.onCreate(b);
setContentView(R.layout.activity_main);


root=(RelativeLayout)findViewById(R.id.root); info=(TextView)findViewById(R.id.info); tap=(Button)findViewById(R.id.tap);

tap.setOnClickListener(new View.OnClickListener(){
public void onClick(View v){ score++;
moveBtn();
}
});


new CountDownTimer(30000,1000){
public void onTick(long ms){
info.setText("Score: "+score+" Time: "+ms/1000);
}
public void onFinish(){
info.setText("Score: "+score+" Time: 0"); tap.setEnabled(false);
}
}.start();
}


void moveBtn(){
int maxX=root.getWidth()-tap.getWidth(); int maxY=root.getHeight()-tap.getHeight(); if(maxX<0)maxX=0;
if(maxY<0)maxY=0;
RelativeLayout.LayoutParams p=(RelativeLayout.LayoutParams)tap.getLayoutParams(); p.leftMargin=r.nextInt(maxX+1);
p.topMargin=r.nextInt(maxY+1); tap.setLayoutParams(p);
}
}
