package com.example.ninth;

 

import android.app.Activity;

import android.os.Bundle;

import android.view.View;

import android.widget.Button;

import android.widget.EditText;

 

// ✅ Use your app's R class (not android.R)

import com.example.ninth.R;

 

public class MainActivity extends Activity implements View.OnClickListener {

 

   EditText n1, n2, res;

   Button btnAdd, btnSub, btnMul, btnDiv;

 

   @Override

   protected void onCreate(Bundle b) {

       super.onCreate(b);

       setContentView(R.layout.activity_main);

 

       // Find views

       n1  = (EditText) findViewById(R.id.etNum1);

       n2  = (EditText) findViewById(R.id.etNum2);

       res = (EditText) findViewById(R.id.etResult);

 

       // Set listeners for all buttons

       int[] ids = { R.id.btnAdd, R.id.btnSub, R.id.btnMul, R.id.btnDiv };

       for (int id : ids) {

           findViewById(id).setOnClickListener(this);

       }

   }

 

   @Override

   public void onClick(View v) {

       calc(v);

   }

 

   void calc(View v) {

       double a = get(n1), b = get(n2), r = 0;

       int id = v.getId();

 

       if (id == R.id.btnAdd) {

           r = a + b;

       } else if (id == R.id.btnSub) {

           r = a - b;

       } else if (id == R.id.btnMul) {

           r = a * b;

       } else if (id == R.id.btnDiv) {

           r = (b != 0) ? a / b : 0;

       }

 

       res.setText(String.valueOf(r));

   }

 

   double get(EditText e) {

       String s = e.getText().toString();

       return s.length() == 0 ? 0 : Double.parseDouble(s);

   }

}
